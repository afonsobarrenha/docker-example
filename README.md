### Dockerfile in Depth
#### RUN
nano Dockerfile
    FROM debian:jessie
    RUN apt-get update && apt-get install -y \
        git \
        vim
docker build -t afonsobarrenha/debian .

#### CMD
nano Dockerfile
    FROM debian:jessian
    RUN apt-get update && apt-get install -y \
        git \
        vim
    CMD ["echo", "hello world"]
docker build -t afonsobarrenha/debian .
docker run image_id
docker run afonsobarrenha/debian echo "hello docker"

#### Docker Cache
docker build -t afonsobarrenha/debian . --no-cache=true

#### COPY
touch abc.txt

    FROM debian:jessian

        git \
